Counter.jsx React 
import React, { useState, useMemo } from 'react';

export default function Counter() {
  const [count, setCount] = useState(0);
  const doubled = useMemo(
    () => count * 2, [count]
  );
  const increment = () => {
    setCount(prev => prev + 1);
  };
  const decrement = () => {
    setCount(prev => prev - 1);
  };

  return (
    <div style={{
      display: 'flex',
      flexDirection: 'column',
      gap: '16px',
      padding: '24px',
      alignItems: 'center',
    }}>
      <span style={{
        fontSize: '32px',
        fontWeight: 'bold',
      }}>Counter</span>
      <span style={{
        fontSize: '40px',
        fontWeight: 'bold',
      }}>{count}</span>

    </div>
  );
}
